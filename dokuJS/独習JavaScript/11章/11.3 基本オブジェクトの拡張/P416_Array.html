Array

配列を操作する

// 配列をクリア
[1, 2, 3].clear() // => []

// 配列からnull、undefinedを除去
[1, 2, null].compact() // => [1, 2]

// ネストした配列をフラット化
[1, [2, 3], 4].flatten() // => [1, 2, 3, 4]

// 逆さにする
[1, 2, 3, 4].reverse(false) // => [4, 3, 2, 1]

// 配列の重複を取り除く
[1, 1, 2].uniq() // => [1, 2]

// 指定した値を取り除く
[1, 2, 3, 4].without(1, 3) // => [2, 4]

配列の要素をイテレータで処理する

var array = ['PHP','Java','JavaScript'];
array.each(function(item, index){
  alert('独習' + item);
});
